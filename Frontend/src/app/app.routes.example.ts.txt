/**
 * Alternative approach: Using lazy-loaded feature modules
 *
 * This is useful for larger applications with many routes
 */

export const alternativeRoutes: Routes = [
  // Public routes
  {
    path: '',
    loadChildren: () => import('./features/public/public.routes').then(m => m.PUBLIC_ROUTES)
  },

  // Authenticated user feature module
  {
    path: 'user',
    canActivate: [authGuard],
    loadChildren: () => import('./features/user/user.routes').then(m => m.USER_ROUTES)
  },

  // Mentor feature module
  {
    path: 'mentor',
    canActivate: [mentorGuard],
    loadChildren: () => import('./features/mentor/mentor.routes').then(m => m.MENTOR_ROUTES)
  },

  // Admin feature module
  {
    path: 'admin',
    canActivate: [adminGuard],
    loadChildren: () => import('./features/admin/admin.routes').then(m => m.ADMIN_ROUTES)
  },

  // Error routes
  {
    path: '**',
    redirectTo: 'not-found'
  }
];
